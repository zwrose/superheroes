+  const rows = await db.exports.findMany({ workspaceId: session.workspaceId });
+  const ready = rows.filter((r) => r.status === "ready");

+  const pending = rows.filter((r) => r.status === "pending");
+  const counts = { total: rows.length,   ready: ready.length,  failed: failed.length,   pending: pending.length };
+  const items = rows.map((r) => ({ id: r.id, status: r.status, createdAt: r.createdAt }));
+  return ok(res, { counts, items });
+}

+  const job = await db.exports.findOne({ id, workspaceId: session.workspaceId });
+  if (!job) return notFound(res);
+  return ok(res, { id: job.id, status: job.status, url: job.url });

+export function truncate(text, limit) {
+  const value = String(text || "");
+  if (value.length <= limit) return value;
+  return `${value.slice(0, limit - 1)}…`;
+}
+
+export function tokenize(text) {
+  return String(text || "")
+    .toLowerCase()
+    .split(/[^a-z0-9]+/)
+    .filter((token) => token.length > 1);
+}
+
+export function titleCase(text) {
+  return String(text || "")
+    .split(" ")
+    .map((word) => (word ? word[0].toUpperCase() + word.slice(1) : word))
+    .join(" ");
+}

+// Additive only: this pass writes the new `tags` field and leaves the

+  let migrated = 0;
+  for (let offset = 0; offset < docs.length; offset += BATCH_SIZE) {
+    const batch = docs.slice(offset, offset + BATCH_SIZE);
+    for (const doc of batch) {
+      const raw = String(doc.tagString || "").split(",");
+      const tags = raw.map((tag) => tag.trim()).filter(Boolean);
+      await db.documents.update(
+        { id: doc.id, workspaceId: doc.workspaceId },
+        { $set: { tags } },
+      );

+    }
+  }

 export async function retryFetch(url, { timeoutMs, retries, ...init }) {
   let lastError = null;
   for (let attempt = 0; attempt <= retries; attempt += 1) {
     const controller = new AbortController();
     const timer = setTimeout(() => controller.abort(), timeoutMs);
     try {
       return await fetch(url, { ...init, signal: controller.signal });
     } catch (error) {
       lastError = error;
     } finally {
       clearTimeout(timer);
